33  /// \brief This class represents the ARM timer used in BCM2835.
34  // -----------------------------------------------------------------------------------------------------------------
35  class CARM_Timer final : public IPeripheral, public ISystem_Clock_Listener
36  {
37  public:
38  // -------------------------------------------------------------------------------------------------------------
39  /// \enum NRegister
40  /// \brief Enumeration of different registers which are used to interact with the timer.
41  // -------------------------------------------------------------------------------------------------------------
42  enum class NRegister : std::uint32_t
43  {
44  Load = 0, ///< The value the timer starts off at (it it loaded into the value register)
45  Value, ///< The current value of the timer (counts down to 0)
46  Control, ///< Control register (see CARM_Timer::TControl_Register)
47  IRQ_Clear, ///< Used to clear a pending IRQ (by writing a 1 to it)
48  IRQ_Raw, ///< When the times reaches zero, this register is set to a 1 regardless of the settings
49  IRQ_Masked, ///< If the timer interrupt is enabled, this register is set to a 1 whenever the timer hits 0
50  Reload, ///< Copy of the load register
51  Pre_Divider, ///< TODO not used?
52  Free_Running, ///< Free running counter (how many times the timer has hit 0)
53  Count ///< Total number of register (helper enum record)
54  };
55

62  // -------------------------------------------------------------------------------------------------------------
63  /// \enum NPrescal_Bits
64  /// \brief Different pre-scaler values that can be used to adjust the speed of the timer.
65  ///
66  /// The timer frequency is inherited from the main CPU clock.
67  // -------------------------------------------------------------------------------------------------------------
68  enum class NPrescal_Bits : std::uint32_t
69  {
70  Prescale_None = 0b00U, ///< No prescaler is present
71  Prescale_16 = 0b01U, ///< The CPU frequency is divided by 16
72  Prescale_256 = 0b10U, ///< The CPU frequency is divided by 256
73  Prescale_1 = 0b11U ///< The CPU frequency is divided by 1 (same as the CPU frequency)
74  };

167  /// It works as wrapper counter around the number of cycles passed in whenever an instruction is executed.
168  // -------------------------------------------------------------------------------------------------------------
169  class CPrescaler final

177  {
178  Prescale_16 = 16U, ///< 16 (the prescaler counts up to 16)
179  Prescale_256 = 256U, ///< 256 (the prescaler counts up to 256)
180  };

182  public:

197  // ---------------------------------------------------------------------------------------------------------
198  std::uint32_t operator()(std::uint32_t cycles_passed) noexcept;
199

212  [[nodiscard]] std::uint32_t Update_Counter(std::uint32_t limit_value);
213 
214  private:

217  };
218 
219  private:
220  // -------------------------------------------------------------------------------------------------------------
221  /// \brief Clear a pending interrupt (helper function).
222  ///
223  /// When a 1 is written to this register, it notifies the interrupt controller, so it does not keep on firing
224  /// an IRQ exception. It also clears the following registers: IRQ_Clear, IRQ_Raw, IRQ_Masked.
225  // -------------------------------------------------------------------------------------------------------------
226  inline void Clear_Basic_IRQ();
227 
228  // -------------------------------------------------------------------------------------------------------------
229  /// \brief This procedure is called whenever the timer reaches 0 (helper function).
230  ///
231  /// It sets the value register to the value stored in the reload register, and sets IRQ_Raw to 1.
232  /// If interrupts are enabled, it also stores a 1 into IRQ_Masked, and signalizes the interrupt controller.
233  /// If the free-running if set to a 1, it increments the Free_Running register.
234  ///
235  /// \param control_reg Reference to the timer control register
236  // -------------------------------------------------------------------------------------------------------------
237  inline void Timer_Has_Reached_Zero(const TControl_Register& control_reg);
