33 /// \class CAUX

35 // -----------------------------------------------------------------------------------------------------------------
36 class CAUX final : public IPeripheral, public ISystem_Clock_Listener
37 {
38 public:

41
42 // -------------------------------------------------------------------------------------------------------------
43 /// \enum NAUX_Peripheral
44 /// \brief This enumeration represents different auxiliary peripherals.
45 // -------------------------------------------------------------------------------------------------------------
46 enum class NAUX_Peripheral : std::uint32_t
47 {
48 Mini_UART = 0, ///< Mini UART
49 SPI_1 = 1, ///< SPI 1
50 SPI_2 = 2, ///< SPI 2
51 Count ///< Total number of auxiliary peripherals
52 };
53
54 // -------------------------------------------------------------------------------------------------------------
55 /// \enum NRegister
56 /// \brief This enumeration represents different register associated with the auxiliaries.
57 // -------------------------------------------------------------------------------------------------------------
58 enum class NRegister : std::uint32_t
59 {
60 IRQ = 0, ///< Pending IRQs
61 ENABLES = 1, ///< Enable different auxiliaries (Mini UART, SPI1, SPI2)
62
63 // Mini UART registers (not all of them are being used)
64
65 MU_IO = 16, ///< IO (sent/received data)
66 MU_IER = 17, ///< IER
67 MU_IIR = 18, ///< IIR
68 MU_LCR = 19, ///< LCR (data length 7/8 bits)
69 MU_MCR = 20, ///< MCR
70 MU_LSR = 21, ///< LSR (transmit FIFO clear)
71 MU_MSR = 22, ///< MSR
72 MU_SCRATCH = 23, ///< SCRATCH
73 MU_CNTL = 24, ///< CNTL (enable transmission/reception)
74 MU_STAT = 25, ///< STAT
75 MU_BAUD = 26, ///< BAUD (baud rate)
76
77 // TODO Add SPI registers
78 Count = 27 ///< Total number of registers (helper record)
79 };
80
