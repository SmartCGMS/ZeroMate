31  /// \class CBus
32  /// \brief An interface the CPU uses to communicate with different peripherals.
33  ///
34  /// It holds a list of all peripherals that have been mapped to the address space.
35  /// When the CPU wants to talk to a peripheral, it must know the address the peripheral
36  /// is mapped to. The CBus class then forwards the data to the corresponding peripheral
37  /// based on this address.
38  // -----------------------------------------------------------------------------------------------------------------
39  class CBus final
40  {
41  public:
42  /// This is just a simplification to make the code less wordy.
43  using Peripheral_t = std::shared_ptr<peripheral::IPeripheral>;

46  /// \enum NStatus
47  /// \brief Return code (status) of mapping a peripheral to the address space.
48  // -------------------------------------------------------------------------------------------------------------
49  enum class NStatus
50  {
51  OK, ///< The peripherals has been mapped successfully
52  Addr_Collision, ///< Another peripheral is already mapped to this region
53  Out_Of_Addr_Space ///< The peripheral does not fit into the address space
54  };

58  /// \struct TMapped_Peripheral
59  /// \brief Representation of a memory-mapped peripheral.
60  // -------------------------------------------------------------------------------------------------------------
61  struct TMapped_Peripheral

63  std::uint32_t start_addr; ///< The starting address of the peripheral
64  Peripheral_t peripheral; ///< The peripheral itself

66  // ---------------------------------------------------------------------------------------------------------
67  /// Compares two memory-mapped peripherals by their addresses
68  ///
69  /// This operator needs to be overloaded, so a binary search can be
70  /// used to figure out what peripheral is being targeted.
71  ///
72  /// \param other Peripheral this peripheral is being compared to
73  /// \return true, if this peripheral is mapped to a lower address than the

98  // -------------------------------------------------------------------------------------------------------------
99  /// \brief Writes data to the given address in the address space.
100  /// \note The bus width is usually a fixed size. The generic type is supported only for emulation purposes
101  /// \tparam Type Data type used when calling the function (determines the number of bytes to be written)
102  /// \param addr Address the data will be written to
103  /// \param value The data itself
104  // -------------------------------------------------------------------------------------------------------------
105  template<typename Type>
106  void Write(std::uint32_t addr, Type value)
107  {
108  // Make calling of the function thread-safe.
109  const std::lock_guard<std::mutex> lock(m_mtx);
110 
111  // Figure out what peripheral is being targeted based on the given address.
112  auto peripheral_iter = Get_Peripheral<Type>(addr);
113 
114  // Calculate the relative address (the peripheral does not have any information
115  // about its location in the address space).
116  const auto relative_addr = addr - peripheral_iter->start_addr;
117 
118  // Write the data to the peripheral.
119  peripheral_iter->peripheral->Write(relative_addr, std::bit_cast<const char*>(&value), sizeof(Type));
120  }
121 
122  // -------------------------------------------------------------------------------------------------------------
123  /// \brief Reads data from the given address in the address space.
124  /// \note The bus width is usually a fixed size. The generic type is supported only for emulation purposes
125  /// \tparam Type Data type used when calling the function (determines the number of bytes to be read)
126  /// \param addr Address the data will be read from
127  /// \return Data read from the given address
128  // -------------------------------------------------------------------------------------------------------------
129  template<typename Type>
130  [[nodiscard]] Type Read(std::uint32_t addr)
131  {
132  // Make calling of the function thread-safe.
133  const std::lock_guard<std::mutex> lock(m_mtx);
134 
135  // Retrieve the peripheral based on the given address.
136  auto peripheral_iter = Get_Peripheral<Type>(addr);
137  const auto relative_addr = addr - peripheral_iter->start_addr;
138 
139  // Read the data from the peripheral.
140  Type value{};
141  peripheral_iter->peripheral->Read(relative_addr, std::bit_cast<char*>(&value), sizeof(Type));
142 
143  return value;
144  }

155  // -------------------------------------------------------------------------------------------------------------
156  /// \brief Checks for unaligned memory access.
157  ///
158  /// If the coprocessor 15 is not present, the function will ignore any unaligned memory access.
159  /// If CP15 is present, it will check if unaligned access has been permitted and validates the
160  /// access accordingly.
161  ///
162  /// \tparam Type Data type used in the memory access
163  /// \param addr Address to be read from or written to
164  /// \return true, if unaligned access takes place. false, otherwise.
165  // -------------------------------------------------------------------------------------------------------------
166  template<typename Type>
167  [[nodiscard]] inline bool Unaligned_Access_Violation(std::uint32_t addr) const
168  {
169  // Make sure CP15 is preset.
170  if (m_cp15 == nullptr)
171  {
172  return false;
173  }
174 
175  // Retrieve the secondary register in which the desired flag is located
176  const auto cp15_c1 =
177  m_cp15->Get_Primary_Register<coprocessor::cp15::CC1>(coprocessor::cp15::NPrimary_Register::C1);
178 
179  return !cp15_c1->Is_Control_Flag_Set(
180  coprocessor::cp15::CC1::NC0_Control_Flags::Unaligned_Memory_Access_Enable) &&
181  (addr % sizeof(Type)) != 0;
182  }
183 
184  // -------------------------------------------------------------------------------------------------------------
185  /// \brief Returns the corresponding peripheral based on the given address.
186  ///
187  /// It searches for a peripheral mapped to the region determined by the given address. It also checks
188  /// memory access alignment, if enabled in CP15.
189  ///
190  /// \tparam Type Data type used when writing or reading from the address space
191  /// \param addr Relative address to be read from or written to
192  /// \return Corresponding peripheral mapped to the region determined by the given address
193  /// \throws CData_Abort if unaligned memory access is detected or no peripheral was mapped into the
194  /// given region
195  // -------------------------------------------------------------------------------------------------------------
196  template<typename Type>
197  [[nodiscard]] Peripherals_t::iterator Get_Peripheral(std::uint32_t addr) const
198  {
199  // Check for unaligned memory access.
200  if (Unaligned_Access_Violation<Type>(addr))
201  {
202  throw arm1176jzf_s::exceptions::CData_Abort{ addr, "Unaligned memory access" };
203  }
204 
205  // Search for a peripheral mapped to the given region.
206  auto peripheral_iter = Get_Peripheral(addr);
207  if (!Is_Peripheral_Accessible(addr, peripheral_iter))
208  {
209  throw arm1176jzf_s::exceptions::CData_Abort{ addr, "No peripheral is mapped to this region" };
210  }
211 
212  return peripheral_iter;
213  }
214 
215  // -------------------------------------------------------------------------------------------------------------
216  /// \brief Checks if a peripheral was found and is accessible.
217  ///
218  /// It checks whether there is a peripheral mapped to the given region in the address space.
219  /// If a peripheral was found, it also checks if the given address exceeds the size of the peripheral
220  /// or not.
221  ///
222  /// \param addr Relative address used to access the peripheral
223  /// \param peripheral_iter Peripheral iterator from a previously applied binary search
224  /// \return true if an accessible peripheral was found in the given region. false otherwise.
225  // -------------------------------------------------------------------------------------------------------------
226  [[nodiscard]] inline bool Is_Peripheral_Accessible(std::uint32_t addr,
227  const Peripherals_t::iterator& peripheral_iter) const
228  {
229  // Make sure the peripheral exists.
230  if (peripheral_iter == m_peripherals.end())
231  {
232  return false;
233  }
234 
235  // Make sure the address does not exceed the size of the peripheral.
236  if ((peripheral_iter->start_addr + peripheral_iter->peripheral->Get_Size()) <= addr)
237  {
238  return false;
239  }
240 
241  return true;
242  }
243 
244  // -------------------------------------------------------------------------------------------------------------
245  /// \brief Binary searches for a peripheral by the given address.
246  ///
247  /// Since peripheral cannot overlap, a binary search can be used to find a peripheral by the given
248  /// address.
249  ///
250  /// \param addr Address in the address space
251  /// \return Peripheral that should be mapped to the given address
252  // -------------------------------------------------------------------------------------------------------------
253  [[nodiscard]] Peripherals_t::iterator Get_Peripheral(std::uint32_t addr) const
254  {
255  // Apply a binary search.
256  // clang-format off
257  auto peripheral_iter = std::lower_bound(
258  m_peripherals.begin(),
259  m_peripherals.end(), addr,
260  [&](const auto& peripheral, std::uint32_t addr_cmp) -> bool {
261  return peripheral.start_addr < addr_cmp;
262  }
263  );
264  // clang-format on
265 
266  // In case we access the very first address of the peripheral, we can simply return it.
267  if (peripheral_iter != m_peripherals.end() && peripheral_iter->start_addr == addr)
268  {
269  return peripheral_iter;
270  }
271 
272  // Otherwise, we have to return the peripheral that was mapped to its immediate left (lower address).
273  if (peripheral_iter != m_peripherals.begin())
274  {
275  --peripheral_iter;
276  }
277 
278  return peripheral_iter;
279  }
