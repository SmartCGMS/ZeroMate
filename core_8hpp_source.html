459  // -------------------------------------------------------------------------------------------------------------
460  /// \brief Calculates the base address of a data transfer instruction.
461  /// \tparam Instruction Type of the instruction that is being executed (block data transfer, RFE, or SRS)
462  /// \param instruction Instruction itself
463  /// \param base_reg_idx Index of the base register (Rd = base address)
464  /// \param cpu_mode CPU mode used for the instruction
465  /// \param number_of_regs Number of registers to be transferred
466  /// \return Base address of the data transfer instruction
467  // -------------------------------------------------------------------------------------------------------------
468  template<typename Instruction>
469  [[nodiscard]] std::uint32_t Calculate_Base_Address(Instruction instruction,
470  std::uint32_t base_reg_idx,
471  CCPU_Context::NCPU_Mode cpu_mode,
472  std::uint32_t number_of_regs) const
473  {
474  switch (instruction.Get_Addressing_Mode())
475  {
476  // Increment before
477  case Instruction::NAddressing_Mode::IB:
478  return m_context.Get_Register(base_reg_idx, cpu_mode) + CCPU_Context::REG_SIZE;
479 
480  // Increment after
481  case Instruction::NAddressing_Mode::IA:
482  return m_context.Get_Register(base_reg_idx, cpu_mode);
483 
484  // Decrement before
485  case Instruction::NAddressing_Mode::DB:
486  return m_context.Get_Register(base_reg_idx, cpu_mode) - (number_of_regs * CCPU_Context::REG_SIZE);
487 
488  // Decrement after
489  case Instruction::NAddressing_Mode::DA:
490  return m_context.Get_Register(base_reg_idx, cpu_mode) - (number_of_regs * CCPU_Context::REG_SIZE) +
491  CCPU_Context::REG_SIZE;
492  }
493 
494  return {}; // Just so the compiler does not gripe about a missing return value.
495  }

497  // -------------------------------------------------------------------------------------------------------------
498  /// \brief Calculates the second immediate operand of a given instruction.
499  /// \tparam Instruction Type of the instruction that is being executed (data processing, MSR)
500  /// \param instruction Instruction itself
501  /// \return Second immediate operand of the instruction
502  // -------------------------------------------------------------------------------------------------------------
503  template<typename Instruction>
504  [[nodiscard]] utils::math::TShift_Result<std::uint32_t>
505  Get_Second_Operand_Imm(Instruction instruction) const noexcept
506  {
507  // Retrieve the immediate value and the shift amount.
508  // The shift amount is always multiplied by 2 - see the documentation.
509  const std::uint32_t immediate = instruction.Get_Immediate();
510  const std::uint32_t shift_amount = instruction.Get_Rotate() * 2;
511 
512  // Create the result.
513  utils::math::TShift_Result<std::uint32_t> second_operand{ m_context.Is_Flag_Set(CCPU_Context::NFlag::C),
514  immediate };
515 
516  // Perform ROR if possible.
517  if (shift_amount != 0 && shift_amount != std::numeric_limits<std::uint32_t>::digits)
518  {
519  second_operand = utils::math::ROR(immediate, shift_amount, false);
520  }
521 
522  return second_operand;
523  }

526  /// \brief Reads/writes data to the bus.
527  /// \note The bus width is usually a fixed size. The generic type is supported only for emulation purposes
528  /// \tparam Type Data type to be read/written to the bus
529  /// \param instruction Instruction that is being executed
530  /// \param addr Address to read or written to
531  /// \param reg_idx Index of the register used in the instruction
532  // -------------------------------------------------------------------------------------------------------------
533  template<std::unsigned_integral Type>
534  void Read_Write_Value(isa::CSingle_Data_Transfer instruction, std::uint32_t addr, std::uint32_t reg_idx)
535  {
536  // Make sure the CPU is connected to the bus, so we can read/write to it.
537  assert(m_bus != nullptr);
538 
539  if (instruction.Is_L_Bit_Set())
540  {
541  // Read data from the given bus address.
542  m_context[reg_idx] = m_bus->Read<Type>(addr);
543  }
544  else
545  {
546  // Write data to the given bus address.
547  m_bus->Write<Type>(addr, static_cast<Type>(m_context[reg_idx]));
548  }
