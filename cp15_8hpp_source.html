57 /// \brief Returns a primary register of a given type.
58 ///
59 /// All primary registers are held by this class as IPrimary_Reg, which they all inherit from. When the caller
60 /// function wants the access a primary register of its actual type, it needs to pass the type as a template.
61 ///
62 /// \tparam Reg_Type Type of the primary register to be returned
63 /// \param primary_reg Index of the primary register to be returned
64 /// \return Primary register of CP15
65 // -------------------------------------------------------------------------------------------------------------
66 template<typename Reg_Type>
67 [[nodiscard]] std::shared_ptr<Reg_Type> Get_Primary_Register(NPrimary_Register primary_reg) const
68 {
69 // Make sure Reg_Type is derived from IPrimary_Reg.
70 static_assert(std::is_base_of<IPrimary_Reg, Reg_Type>::value);
71
72 // Return the primary register of a desired type.
73 return std::static_pointer_cast<Reg_Type>(m_regs.at(static_cast<std::uint32_t>(primary_reg)));
74 }
