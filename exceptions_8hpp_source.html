30 /// \class CCPU_Exception
31 /// \brief This class represents a parent class for all the CPU exceptions.
32 // -----------------------------------------------------------------------------------------------------------------
33 class CCPU_Exception : public std::runtime_error
34 {
35 public:
36 /// High base address of the IVT table (reallocation is enabled in CP15)
37 static constexpr std::uint32_t IVT_High_Base_Addr = 0xFFFF0000;
38
39 // -------------------------------------------------------------------------------------------------------------
40 /// \enum NType
41 /// \brief Type of the exception.
42 // -------------------------------------------------------------------------------------------------------------
43 enum class NType
44 {
45 Reset, ///< Reset the CPU
46 Undefined_Instruction, ///< Undefined instruction
47 Software_Interrupt, ///< SWI (software interrupt; used for syscalls)
48 Prefetch_Abort, ///< Prefetch abort (the CPU failed to fetch the next instruction)
49 Data_Abort, ///< Data abort when reading/writing to the bus
50 IRQ, ///< Interrupt
51 FIQ ///< Fast interrupt
52 };
53
54 public:
55 // -------------------------------------------------------------------------------------------------------------
56 /// \brief Constructor of the class.
57 /// \param exception_vector Address of the handler of the exception (IVT = interrupt vector table)
58 /// \param mode Mode into which the CPU will switch when the exception is thrown
59 /// \param name Description of the exception for logging purposes
60 /// \param type Type of the exception
61 // -------------------------------------------------------------------------------------------------------------
62 explicit CCPU_Exception(std::uint32_t exception_vector,
63 CCPU_Context::NCPU_Mode mode,
64 const char* name,
65 NType type);
66
67 // -------------------------------------------------------------------------------------------------------------
68 /// \brief Returns the address of the handler of the exception.
69 /// \return Address of the handler (IVT = interrupt vector table)
70 // -------------------------------------------------------------------------------------------------------------
71 [[nodiscard]] std::uint32_t Get_Exception_Vector() const;
72
73 // -------------------------------------------------------------------------------------------------------------
74 /// \brief Returns the mode into which the CPU will switch when the exception is thrown.
75 /// \return Mode of the exception.
76 // -------------------------------------------------------------------------------------------------------------
77 [[nodiscard]] CCPU_Context::NCPU_Mode Get_CPU_Mode() const;
78
79 // -------------------------------------------------------------------------------------------------------------
80 /// \brief Returns the type of the exception
81 /// \return Exception type
82 // -------------------------------------------------------------------------------------------------------------
83 [[nodiscard]] NType Get_Type() const;
84
85 protected:
86 std::uint32_t m_exception_vector; ///< Address of the exception handler (IVT = interrupt vector table)
87 CCPU_Context::NCPU_Mode m_mode; ///< Mode into which the CPU will switch when the exception is thrown
88 NType m_type; ///< Type of the exception
89 };
