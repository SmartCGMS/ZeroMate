182  /// \enum NRegister
183  /// \brief Enumeration of different registers which are used to interact with the GPIO pins.
184  // -------------------------------------------------------------------------------------------------------------
185  enum class NRegister : std::uint32_t
186  {
187  GPFSEL0 = 0, ///< Function of GPIO pins 0-9 (3 bits per pin)
188  GPFSEL1, ///< Function of GPIO pins 10-19 (3 bits per pin)
189  GPFSEL2, ///< Function of GPIO pins 02-29 (3 bits per pin)
190  GPFSEL3, ///< Function of GPIO pins 30-39 (3 bits per pin)
191  GPFSEL4, ///< Function of GPIO pins 40-49 (3 bits per pin)
192  GPFSEL5, ///< Function of GPIO pins 50-23 (3 bits per pin)
193  Reserved_01, ///< Reserved
194  GPSET0, ///< Sets the value of a GPIO pin (pins 0-31)
195  GPSET1, ///< Sets the value of a GPIO pin (pins 32-53)
196  Reserved_02, ///< Reserved
197  GPCLR0, ///< Clears the value of a GPIO pin (pins 0-31)
198  GPCLR1, ///< Clears the value of a GPIO pin (pins 32-53)
199  Reserved_03, ///< Reserved
200  GPLEV0, ///< Actual value of the pin (pins 0-31)
201  GPLEV1, ///< Actual value of the pin (pins 32-53)
202  Reserved_04, ///< Reserved
203  GPEDS0, ///< GPIO Pin Event Detect Status 0 (pins 0-31)
204  GPEDS1, ///< GPIO Pin Event Detect Status 1 (pins 32-53)
205  Reserved_05, ///< Reserved
206  GPREN0, ///< Enables/disables the rising edge interrupt for pins 0-31
207  GPREN1, ///< Enables/disables the rising edge interrupt for pins 32-53
208  Reserved_06, ///< Reserved
209  GPFEN0, ///< Enables/disables the falling edge interrupt for pins 0-31
210  GPFEN1, ///< Enables/disables the falling edge interrupt for pins 32-53
211  Reserved_07, ///< Reserved
212  GPHEN0, ///< Enables/disables the high value interrupt for pins 0-31
213  GPHEN1, ///< Enables/disables the high value interrupt for pins 32-53
214  Reserved_08, ///< Reserved
215  GPLEN0, ///< Enables/disables the low value interrupt for pins 0-31
216  GPLEN1, ///< Enables/disables the low value interrupt for pins 32-53
217  Reserved_09, ///< Reserved
218  GPAREN0, ///< Enables/disables the asynchronous rising edge interrupt for pins 0-31
219  GPAREN1, ///< Enables/disables the asynchronous rising edge interrupt for pins 32-53
220  Reserved_10, ///< Reserved
221  GPAFEN0, ///< Enables/disables the asynchronous falling edge interrupt for pins 0-31
222  GPAFEN1, ///< Enables/disables the asynchronous falling edge interrupt for pins 32-53
223  Reserved_11, ///< Reserved
224  GPPUD, ///< Pull-up/down register controls
225  GPPUDCLK0, ///< Pull-up/down clock register 0
226  GPPUDCLK1, ///< Pull-up/down clock register 1
227  Reserved_12, ///< Reserved
228  Test, ///< Test
229  Count ///< Total number of registers (helper enum record)
230  };
