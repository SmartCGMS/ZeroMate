42  LO = 0b0011, ///< C clear; unsigned lower
43  MI = 0b0100, ///< N set; negative
44  PL = 0b0101, ///< N clear; positive or zero
45  VS = 0b0110, ///< V set; overflow
46  VC = 0b0111, ///< V clear; no overflow
47  HI = 0b1000, ///< C set and Z clear; unsigned higher

52  LE = 0b1101, ///< Z set OR (N not equal to V); less than or equal
53  AL = 0b1110, ///< always
54  Unconditioned = 0b1111 ///< always (condition is not applied)
55  };
