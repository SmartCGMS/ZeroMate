29 /// \class CInterrupt_Controller
30 /// \brief This class represents the interrupt controller used in BCM2835.
31 // -----------------------------------------------------------------------------------------------------------------
32 class CInterrupt_Controller final : public IPeripheral
33 {
34 public:
35 // -------------------------------------------------------------------------------------------------------------
36 /// \enum NRegister
37 /// \brief This enumeration lists out all interrupt control registers.
38 // -------------------------------------------------------------------------------------------------------------
39 enum class NRegister : std::uint32_t
40 {
41 IRQ_Basic_Pending = 0, ///< Bitmask of pending basic IRQs
42 IRQ_Pending_1, ///< Bitmask of pending shared IRQs 0-31
43 IRQ_Pending_2, ///< Bitmask of pending shared IRQs 32-63
44 FIQ_Control, ///< Fast interrupt control
45 Enable_IRQs_1, ///< Write 1 to the corresponding bit(s) to enable one or more IRQs (0-31)
46 Enable_IRQs_2, ///< Write 1 to the corresponding bit(s) to enable one or more IRQs (32-63)
47 Enable_Basic_IRQs, ///< Write 1 to the corresponding bit(s) to enable one or more ARM-specific IRQs
48 Disable_IRQs_1, ///< Write 1 to the corresponding bit(s) to disable one or more IRQs (0-31)
49 Disable_IRQs_2, ///< Write 1 to the corresponding bit(s) to disable one or more IRQs (32-63)
50 Disable_Basic_IRQs, ///< Write 1 to the corresponding bit(s) to disable one or more ARM-specific IRQs
51 Count ///< Total number of register (helper enum record)
52 };
53
54 // -------------------------------------------------------------------------------------------------------------
55 /// \enum NIRQ_Basic_Source
56 /// \brief This enumeration defines different IRQ basic source that can trigger an interrupt.
57 ///
58 /// Each value represents a bit position in the corresponding enable/disable register.
59 // -------------------------------------------------------------------------------------------------------------
60 enum class NIRQ_Basic_Source : std::uint32_t
61 {
62 ARM_Timer = 0, ///< ARM timer
63 ARM_Mailbox = 1, ///< Mailbox
64 ARM_Doorbell_0 = 2, ///< Doorbell 0
65 ARM_Doorbell_1 = 3, ///< Doorbell 0
66 GPU_0_Halted = 4, ///< GPU halted 0
67 GPU_1_Halted = 5, ///< GPU halted 0
68 Illegal_Access_Type_1 = 6, ///< Illegal access type 1
69 Illegal_Access_Type_0 = 7, ///< Illegal access type 0
70 };
71
72 // -------------------------------------------------------------------------------------------------------------
73 /// \enum NIRQ_Source
74 /// \brief This enumeration defines different IRQ source that can trigger an interrupt.
75 ///
76 /// Each value represents a bit position in the corresponding enable/disable register. If the position is
77 /// greater than 31, register (2) is used.
78 // -------------------------------------------------------------------------------------------------------------
79 enum class NIRQ_Source
80 {
81 AUX = 29, ///< Auxiliary interrupt
82 I2C_SPI_SLV = 43, ///< I2C or SPI slave
83 PWA_0 = 45, ///< PWA 0
84 PWA_1 = 45, ///< PWA 1
85 SMI = 48, ///< SMI
86 GPIO_0 = 49, ///< BANK 0 GPIO pins 0-27
87 GPIO_1 = 50, ///< BANK 1 GPIO pins 28-45
88 GPIO_2 = 51, ///< BANK 2 GPIO pins 46-53
89 GPIO_3 = 52, ///< All GPIO pins
90 I2C = 53, ///< I2C (master?)
91 SPI = 54, ///< SPI (master?)
92 PCM = 55, ///< PCM
93 UART = 57 ///< UART
94 };
95
96 /// Total number of registers associated with the peripheral
97 static constexpr auto Number_Of_Registers = static_cast<std::uint32_t>(NRegister::Count);
98
99 /// Size of a single register
100 static constexpr auto Reg_Size = static_cast<std::uint32_t>(sizeof(std::uint32_t));
101
102 // -------------------------------------------------------------------------------------------------------------
103 /// \struct TInterrupt_Info
104 /// \brief This structure holds information about each IRQ/Basic IRQ source.
105 // -------------------------------------------------------------------------------------------------------------
106 struct TInterrupt_Info
107 {
108 bool enabled{ false }; ///< Is the source enabled or disabled?
109 bool pending{ false }; ///< Is the source pending or not?
110 };
111
112 /// Default values of the TInterrupt_Info structure
113 static constexpr TInterrupt_Info Disabled_Non_Pending_Interrupt{ .enabled = false, .pending = false };
114
115 public:
116 // -------------------------------------------------------------------------------------------------------------
117 /// \brief Creates an instance of the class.
118 /// \param cpu_context CPU context (the IC needs to check if interrupts are globally enabled/disabled)
119 // -------------------------------------------------------------------------------------------------------------
120 explicit CInterrupt_Controller(const arm1176jzf_s::CCPU_Context& cpu_context);
121
122 // -------------------------------------------------------------------------------------------------------------
123 /// \brief Resets/re-initializes the interrupt controller (IPeripheral interface).
124 // -------------------------------------------------------------------------------------------------------------
125 void Reset() noexcept override;
126
127 // -------------------------------------------------------------------------------------------------------------
128 /// \brief Returns the size of the peripheral (IPeripheral interface).
129 /// \return number of register * register size
130 // -------------------------------------------------------------------------------------------------------------
131 [[nodiscard]] std::uint32_t Get_Size() const noexcept override;
132
133 // -------------------------------------------------------------------------------------------------------------
134 /// \brief Writes data to the peripheral (IPeripheral interface).
135 /// \param addr Relative address (from the peripheral's perspective) where the data will be written
136 /// \param data Pointer to the data to be written to the peripheral
137 /// \param size Size of the data to be written to the peripheral [B]
138 // -------------------------------------------------------------------------------------------------------------
139 void Write(std::uint32_t addr, const char* data, std::uint32_t size) override;
140
141 // -------------------------------------------------------------------------------------------------------------
142 /// \brief Reads data from the peripheral (IPeripheral interface).
143 /// \param addr Relative address (from the peripheral's perspective) from which the data will be read
144 /// \param data Pointer to a buffer the data will be copied into
145 /// \param size Size of the data to read from the peripheral [B]
146 // -------------------------------------------------------------------------------------------------------------
147 void Read(std::uint32_t addr, char* data, std::uint32_t size) override;
148
149 // -------------------------------------------------------------------------------------------------------------
150 /// \brief Notifies the interrupt controller about a pending interrupt triggered by an IRQ source.
151 ///
152 /// The interrupt controller may ignore this notification if the source is not enabled or all interrupts
153 /// have been globally disabled in the CPSR register.
154 ///
155 /// \param source IRQ source that has triggered the interrupt
156 // -------------------------------------------------------------------------------------------------------------
157 void Signalize_IRQ(NIRQ_Source source);
158
159 // -------------------------------------------------------------------------------------------------------------
160 /// \brief Notifies the interrupt controller about a pending interrupt triggered by a basic IRQ source.
161 ///
162 /// The interrupt controller may ignore this notification if the source is not enabled or all interrupts
163 /// have been globally disabled in the CPSR register.
164 ///
165 /// \param source Basic IRQ source that has triggered the interrupt
166 // -------------------------------------------------------------------------------------------------------------
167 void Signalize_Basic_IRQ(NIRQ_Basic_Source source);
168
169 // -------------------------------------------------------------------------------------------------------------
170 /// \brief Returns information about whether there is a pending IRQ.
171 ///
172 /// For this function to return true, all interrupts must be globally enabled in the CPSR register.
173 /// Additionally, there must be a basic IRQ pending or an IRQ pending (see #Has_Pending_IRQ, and
174 /// #Has_Pending_Basic_IRQ).
175 ///
176 /// \return true, if there is an pending interrupt. false, otherwise.
177 // -------------------------------------------------------------------------------------------------------------
178 [[nodiscard]] bool Has_Pending_Interrupt() const noexcept;
179
180 // -------------------------------------------------------------------------------------------------------------
181 /// \brief Clears a pending basic IRQ.
182 ///
183 /// This function is called from the corresponding peripheral whenever it gets cleared of a pending interrupt
184 /// by writing a 1 to its clear register.
185 ///
186 /// \param source Basic IRQ source that has a pending interrupt.
187 // -------------------------------------------------------------------------------------------------------------
188 void Clear_Pending_Basic_IRQ(NIRQ_Basic_Source source) noexcept;
189
190 // -------------------------------------------------------------------------------------------------------------
191 /// \brief Clears a pending IRQ.
192 ///
193 /// This function is called from the corresponding peripheral whenever it gets cleared of a pending interrupt
194 /// by writing a 1 to its clear register.
195 ///
196 /// \param source IRQ source that has a pending interrupt.
197 // -------------------------------------------------------------------------------------------------------------
198 void Clear_Pending_IRQ(NIRQ_Source source) noexcept;
199
200 // -------------------------------------------------------------------------------------------------------------
201 /// \brief Returns a collection of basic IRQs (visualization purposes).
202 /// \return Collection of basic IRQs
203 // -------------------------------------------------------------------------------------------------------------
204 [[nodiscard]] const std::unordered_map<NIRQ_Basic_Source, TInterrupt_Info>& Get_Basic_IRQs() const;
205
206 // -------------------------------------------------------------------------------------------------------------
207 /// \brief Returns a collection of IRQs (visualization purposes).
208 /// \return Collection of IRQs
209 // -------------------------------------------------------------------------------------------------------------
210 [[nodiscard]] const std::unordered_map<NIRQ_Source, TInterrupt_Info>& Get_IRQs() const;
211
212 // -------------------------------------------------------------------------------------------------------------
213 /// \brief Returns the corresponding IRQ source for a given GPIO pin.
214 ///
215 /// - gpio_int[0] for BANK0 (pins 0-27)
216 /// - gpio_int[1] for BANK1 (pins 28-45)
217 /// - gpio_int[2] for BANK2 (pins 46-53)
218 /// - gpio_int[3] for all the pins; visit
219 /// https://raspberrypi.stackexchange.com/questions/51737/use-of-gpio-interrupt-2-rather-than-interrupt-3.
220 ///
221 /// \param pin_idx Index of the GPIO pin
222 /// \return Corresponding IRQ source (GPIO_0-3)
223 // -------------------------------------------------------------------------------------------------------------
224 [[nodiscard]] static NIRQ_Source Get_IRQ_Source(std::size_t pin_idx) noexcept;
225
226 private:
227 // -------------------------------------------------------------------------------------------------------------
228 /// \brief Initializes the collection of basic IRQ sources.
229 // -------------------------------------------------------------------------------------------------------------
230 inline void Initialize_Basic_IRQ_Sources();
231
232 // -------------------------------------------------------------------------------------------------------------
233 /// \brief Initializes the collection of IRQ sources.
234 // -------------------------------------------------------------------------------------------------------------
235 inline void Initialize_IRQ_Sources();
236
237 // -------------------------------------------------------------------------------------------------------------
238 /// \brief Checks whether an IRQ source is enabled or not.
239 /// \param source IRQ source to be examined
240 /// \return true, if the IRQ source is enabled. false, otherwise.
241 // -------------------------------------------------------------------------------------------------------------
242 [[nodiscard]] inline bool Is_IRQ_Source_Enabled(NIRQ_Source source) const;
243
244 // -------------------------------------------------------------------------------------------------------------
245 /// \brief Checks whether there is a pending interrupt triggered by an IRQ source.
246 /// \return true, if there is a pending interrupt (IRQ source). false, otherwise.
247 // -------------------------------------------------------------------------------------------------------------
248 [[nodiscard]] inline bool Has_Pending_IRQ() const;
249
250 // -------------------------------------------------------------------------------------------------------------
251 /// \brief Checks whether there is a pending interrupt triggered by a basic IRQ source.
252 /// \return true, if there is a pending interrupt (basic IRQ source). false, otherwise.
253 // -------------------------------------------------------------------------------------------------------------
254 [[nodiscard]] inline bool Has_Pending_Basic_IRQ() const;
255
256 // -------------------------------------------------------------------------------------------------------------
257 /// \brief Enables/disables interrupts from a basic IRQ source.
258 /// \param reg Register containing information about what basic IRQ source are enabled/disabled
259 /// \param enable Indication of whether the register is an enable or disable register
260 // -------------------------------------------------------------------------------------------------------------
261 void Enable_IRQ_Basic_Sources(NRegister reg, bool enable);
262
263 // -------------------------------------------------------------------------------------------------------------
264 /// \brief Enables/disables interrupts from an IRQ source.
265 /// \param reg Register containing information about what IRQ source are enabled/disabled
266 /// \param enable Indication of whether the register is an enable or disable register
267 // -------------------------------------------------------------------------------------------------------------
268 void Enable_IRQ_Sources(NRegister reg, bool enable);
269
270 private:
271 const arm1176jzf_s::CCPU_Context& m_cpu_context; ///< CPU context
272 std::array<std::uint32_t, Number_Of_Registers> m_regs; ///< IC registers
273 std::unordered_map<NIRQ_Basic_Source, TInterrupt_Info> m_irq_basic_sources; ///< Collection of IRQ basic sources
274 std::unordered_map<NIRQ_Source, TInterrupt_Info> m_irq_sources; ///< Collection of IRQ sources
275 utils::CLogging_System& m_logging_system; ///< Logging system
276 };
