29 /// \class CCircular_Log_Buffer
30 /// \brief This class represents a circular buffer (for log messages)
31 /// \tparam Type Type of a log message (e.g. std::string)
32 /// \tparam Size Size of the buffer (maximum number of elements it can hold at a time)
33 // -----------------------------------------------------------------------------------------------------------------
34 template<typename Type, std::size_t Size>
35 class CCircular_Log_Buffer final
36 {
37 public:
38 // -------------------------------------------------------------------------------------------------------------
39 /// \brief Creates an instance of the class.
40 // -------------------------------------------------------------------------------------------------------------
41 CCircular_Log_Buffer() = default;
42
43 // -------------------------------------------------------------------------------------------------------------
44 /// \brief Adds a log message into the buffer.
45 /// \param value Log message to be added into the circular buffer.
46 // -------------------------------------------------------------------------------------------------------------
47 void Add(Type value)
48 {
49 // Check if the maximum number of elements has been exceeded.
50 if (m_data.size() >= Size)
51 {
52 // Remove the first element.
53 m_data.erase(m_data.begin());
54 }
55
56 m_data.push_back(value);
57 }
58
59 // -------------------------------------------------------------------------------------------------------------
60 /// \brief Clears the circular buffer (deletes all log messages).
61 // -------------------------------------------------------------------------------------------------------------
62 void Clear()
63 {
64 m_data.clear();
65 }
66
67 // -------------------------------------------------------------------------------------------------------------
68 /// \brief Returns a collection of log messages currently stored in the buffer.
69 /// \return Collection of log messages.
70 // -------------------------------------------------------------------------------------------------------------
71 [[nodiscard]] const auto& Get_Data() const
72 {
73 return m_data;
74 }
75
76 private:
77 std::vector<Type> m_data; ///< Collection of log messages
78 };
