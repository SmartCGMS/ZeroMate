89  {
90  // Make sure that Type is made up of more bits than Type_Narrower.
91  static_assert(std::numeric_limits<Type_Narrower>::digits <= std::numeric_limits<Type>::digits);
92

109  /// \brief Performs an LSL (logical shift left) operation on the given integral value.
110  ///
111  /// The following tool can be used to see how these kinds of shifts/rotations work:
112  /// https://onlinetoolz.net/bitshift. Their implementation is also explained in the ARM instruction set manual:
113  /// https://iitd-plos.github.io/col718/ref/arm-instructionset.pdf (section 4.5.2 - Shifts)
114  ///
115  /// \tparam Type Type of the variable the function is called with
116  /// \param value Value used in the operation
117  /// \param shift_size Number of positions the value will be shifted by
118  /// \param carry_flag Carry flag from the CPSR register
119  /// \return TShift_Result, result of the shift operation
120  // -----------------------------------------------------------------------------------------------------------------
121  template<std::unsigned_integral Type>
122  [[nodiscard]] TShift_Result<Type> LSL(Type value, Type shift_size, bool carry_flag) noexcept
123  {
124  bool updated_carry_flag{ false }; // New value of the carry flag (after the operation)
125  Type result{ 0 };
126 
127  if (shift_size == 0)
128  {
129  updated_carry_flag = carry_flag;
130  result = value;
131  }
132  else
133  {
134  updated_carry_flag = Is_Bit_Set<Type>(value, (std::numeric_limits<Type>::digits - shift_size));
135 
136  // Make sure it is safe to perform the shift.
137  if (shift_size < std::numeric_limits<Type>::digits)
138  {
139  result = static_cast<Type>(value << shift_size);
140  }
141  }
142 
143  return { updated_carry_flag, result };
144  }

185  ///
186  /// The following tool can be used to see how these kinds of shifts/rotations work:
187  /// https://onlinetoolz.net/bitshift. Their implementation is also explained in the ARM instruction set manual:
188  /// https://iitd-plos.github.io/col718/ref/arm-instructionset.pdf (section 4.5.2 - Shifts)
189  ///
190  /// \tparam Type Type of the variable the function is called with
191  /// \param value Value used in the operation
192  /// \param shift_size Number of positions the value will be shifted by
193  /// \return TShift_Result, result of the shift operation
194  // -----------------------------------------------------------------------------------------------------------------
195  template<std::unsigned_integral Type>
196  [[nodiscard]] TShift_Result<Type> ASR(Type value, Type shift_size) noexcept
197  {
198  bool carry_flag{ false };
199  Type result{ 0 };
200 
201  if (shift_size == 0)
202  {
203  bool is_negative = Is_Bit_Set<Type>(value, std::numeric_limits<Type>::digits - 1U);
204  carry_flag = is_negative;
205 
206  if (is_negative)
207  {
208  result = static_cast<Type>(-1);
209  }
210  else
211  {
212  result = value;
213  }
214  }
215  else
216  {
217  carry_flag = Is_Bit_Set<Type>(value, (shift_size - 1));
218 
219  // Make sure it is safe to perform the shift.
220  if (shift_size < std::numeric_limits<Type>::digits)
221  {
222  result = static_cast<Type>(static_cast<std::make_signed<Type>::type>(value) >> shift_size);
223  }
224  }
225 
226  return { carry_flag, result };

231  ///
232  /// The following tool can be used to see how these kinds of shifts/rotations work:
233  /// https://onlinetoolz.net/bitshift. Their implementation is also explained in the ARM instruction set manual:
234  /// https://iitd-plos.github.io/col718/ref/arm-instructionset.pdf (section 4.5.2 - Shifts)
235  ///
236  /// \tparam Type Type of the variable the function is called with
237  /// \param value Value used in the operation
238  /// \param shift_size Number of positions the value will be rotated by
239  /// \param carry_flag Carry flag from the CPSR register
240  /// \return TShift_Result, result of the shift operation
241  // -----------------------------------------------------------------------------------------------------------------
242  template<std::unsigned_integral Type>
243  [[nodiscard]] TShift_Result<Type> ROR(Type value, Type shift_size, bool carry_flag) noexcept
244  {
245  bool updated_carry_flag{ false };
246  Type result{ 0 };
247 
248  if (shift_size == 0)
249  {
250  updated_carry_flag = Is_Bit_Set<Type>(value, 0);
251  result = static_cast<Type>((static_cast<Type>(carry_flag) << (std::numeric_limits<Type>::digits - 1U))) |
252  (value >> 1U);
253  }
254  else
255  {
256  updated_carry_flag = Is_Bit_Set<Type>(value, (shift_size - 1));
257 
258  result = static_cast<Type>(value >> shift_size) |
259  static_cast<Type>(value << (std::numeric_limits<Type>::digits - shift_size));
260  }
261 
262  return { updated_carry_flag, result };
263  }

285  /// \tparam Type Type of the two operands the function is called with
286  /// \param op1 First operand
287  /// \param op2 Second operand
288  /// \return true, if subtracting the two operands causes an overflow or underflow. false, if it is safe to perform
289  /// the operation.

291  template<std::signed_integral Type>
292  [[nodiscard]] bool Check_Overflow_Subtraction(Type op1, Type op2) noexcept
293  {
294  if (op2 < 0 && op1 > (std::numeric_limits<Type>::max() + op2))
295  {
296  return true; // overflow
297  }
298  if (op2 > 0 && op1 < (std::numeric_limits<Type>::lowest() + op2))
299  {
300  return true; // underflow
301  }
302 
303  return false;
304  }

330  /// \brief Checks if adding or subtracting the two operands (including the carry flag) causes an overflow/underflow.
331  /// \tparam Type Type of the two operands
332  /// \param op1 First operand
333  /// \param op2 Second operand
334  /// \param subtraction Flag indicating whether the operands should be subtracted or added up
335  /// \param carry Carry flag
336  /// \return true, if the operation causes an overflow/underflow. false, otherwise.
337  // -----------------------------------------------------------------------------------------------------------------
338  template<std::signed_integral Type>
339  [[nodiscard]] bool Check_Overflow(Type op1, Type op2, bool subtraction, Type carry) noexcept
340  {
341  // Check if adding the carry bit would cause an overflow.
342  if (Check_Overflow_Addition<Type>(op2, carry))
343  {
344  return true;
345  }
346 
347  // Add the carry bit.
348  op2 += carry;
349 
350  if (subtraction)
351  {
352  return Check_Overflow_Subtraction<Type>(op1, op2);
353  }
354 
355  return Check_Overflow_Addition<Type>(op1, op2);
356  }

359  /// \brief Sign-extends the given integral value.
360  ///
361  /// If the value is positive, it is simple extended to the large datatype. If the is negative, the final result is
362  /// made up of all 0xF, except for the least significant bits which hold the representation of the given value.
363  ///
364  /// \tparam Small_Type Type of the value passed in as a parameter
365  /// \tparam Large_Type Type to which the value will be sign-extended
366  /// \param value Value to be sign-extended
367  /// \return sign-extended value
368  // -----------------------------------------------------------------------------------------------------------------
369  template<std::unsigned_integral Small_Type, std::unsigned_integral Large_Type = std::uint32_t>
370  [[nodiscard]] Large_Type Sign_Extend_Value(Small_Type value) noexcept
371  {
372  // Make sure Small_Type consists of fewer bits than Large_Type.
373  static_assert(std::numeric_limits<Small_Type>::digits < std::numeric_limits<Large_Type>::digits);
374 
375  if (utils::math::Is_Negative<Small_Type>(value))
376  {
377  // mask = 0xFFFF...
378  auto mask = static_cast<Large_Type>(-1);
379 
380  // Create room for the value by shifting the mask to the left by 8 (adding zeros)
381  for (std::size_t i = 0; i < sizeof(Small_Type); ++i)
382  {
383  mask =
384  static_cast<Large_Type>(mask << static_cast<Large_Type>(std::numeric_limits<std::uint8_t>::digits));
385  }
386 
387  // Insert the value into the final value (modifies the least significant bits).
388  return static_cast<Large_Type>(mask | static_cast<Large_Type>(value));
389  }
390 
391  // If the value is not negative, we can simply cast it.
392  return static_cast<Large_Type>(value);
393  }
