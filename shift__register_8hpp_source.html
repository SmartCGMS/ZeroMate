143  // -------------------------------------------------------------------------------------------------------------
144  /// \brief Initializes GPIO pin subscription.
145  // -------------------------------------------------------------------------------------------------------------
146  inline void Init_GPIO_Subscription()
147  {
148  // Clear the collection (re-initialization).
149  m_gpio_subscription.clear();
150 
151  // Subscribe (listen) to the following pins.
152  m_gpio_subscription.insert(m_data_pin_idx);
153  m_gpio_subscription.insert(m_clock_pin_idx);
154  m_gpio_subscription.insert(m_latch_pin_idx);
155  }

157  // -------------------------------------------------------------------------------------------------------------
158  /// \brief Handles the change of the state of the clock pin.
159  /// \param pin_idx Index of the GPIO pin whose state has changed
160  // -------------------------------------------------------------------------------------------------------------
161  inline void Handle_Clock_Signal(std::uint32_t pin_idx)
162  {
163  // Make sure the pin index is the clock pin.
164  if (pin_idx != m_clock_pin_idx)
165  {
166  return;
167  }
168 
169  // Read the value of the GPIO clock pin.
170  m_clock_state = m_gpio_manager->Read_GPIO_Pin(m_clock_pin_idx);
171 
172  // Check if the clock signal has gone from High to Low.
173  if (m_clock_state_prev == CGPIO_Manager::CPin::NState::High &&
174  m_clock_state == CGPIO_Manager::CPin::NState::Low)
175  {
176  // Shift the register one bit to the right.
177  m_value >>= 1;
178 
179  // Add the input bit to the far right.
180  m_value |= static_cast<Register>((m_last_input_value << (std::numeric_limits<Register>::digits - 1)));
181 
182  // Clear the input of the shift register.
183  m_last_input_value = 0;
184  }
185 
186  // Update the previous state of the pin.
187  m_clock_state_prev = m_clock_state;
188  }
189 
190  // -------------------------------------------------------------------------------------------------------------
191  /// \brief Handles the change of the state of the latch pin.
192  /// \param pin_idx Index of the GPIO pin whose state has changed
193  // -------------------------------------------------------------------------------------------------------------
194  inline void Handle_Latch_Signal(std::uint32_t pin_idx)
195  {
196  // Make sure the pin index is the latch pin.
197  if (pin_idx != m_latch_pin_idx)
198  {
199  return;
200  }
201 
202  // Read the value of the GPIO latch pin.
203  if (m_gpio_manager->Read_GPIO_Pin(m_latch_pin_idx) == CGPIO_Manager::CPin::NState::High)
204  {
205  // Propagate the value to the output of the register.
206  m_output_value = m_value;
207  }
208  }
209 
210  // -------------------------------------------------------------------------------------------------------------
211  /// \brief Handles the change of the state of the data pin.
212  /// \param pin_idx Index of the GPIO pin whose state has changed
213  // -------------------------------------------------------------------------------------------------------------
214  inline void Handle_Data_Signal(std::uint32_t pin_idx)
215  {
216  // Make sure the pin index is the data pin.
217  if (pin_idx != m_data_pin_idx)
218  {
219  return;
220  }
221 
222  // Read the state of the GPIO data pin and store it as the last input value to the shift register.
223  const auto data_pin_value = m_gpio_manager->Read_GPIO_Pin(m_data_pin_idx);
224  m_last_input_value = static_cast<std::uint32_t>(data_pin_value);
225  }
226

228  std::shared_ptr<CGPIO_Manager> m_gpio_manager; ///< GPIO manager
229  std::uint32_t m_last_input_value; ///< Last input value
230  Register m_value; ///< Current value of the shift register
231  Register m_output_value; ///< Output value of the shift register (latch is enabled)
232  std::uint32_t m_latch_pin_idx; ///< Index of the GPIO latch pin
233  std::uint32_t m_data_pin_idx; ///< Index of the GPIO data pin
234  std::uint32_t m_clock_pin_idx; ///< Index of the GPIO clock pin
235  CGPIO_Manager::CPin::NState m_clock_state; ///< State of the clock pin
236  CGPIO_Manager::CPin::NState m_clock_state_prev; ///< Previous state of the clock pin
237  };
