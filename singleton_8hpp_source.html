20 /// \class CSingleton
21 /// \brief A generic class that allows for treating other classes as singletons.
22 ///
23 /// If a class is meant to be treated as a singleton using CSingleton, it needs to implement and empty
24 /// constructor, so CSingleton can instantiate it. CSingleton works as a proxy that provides access to
25 /// the instance of the class.
26 ///
27 /// \tparam Type class to be treated as a singleton
28 // -----------------------------------------------------------------------------------------------------------------
29 template<typename Type>
30 class CSingleton
31 {
32 public:
33 // -------------------------------------------------------------------------------------------------------------
34 /// \brief Returns the instance of the class
35 ///
36 /// The class gets instantiated only once.
37 ///
38 /// \return the instance of the class
39 // -------------------------------------------------------------------------------------------------------------
40 static Type* Get_Instance()
41 {
42 // Make sure the class gets instantiated only once.
43 std::call_once(s_init_flag, [=]() -> void { CSingleton<Type>::s_instance = std::make_unique<Type>(); });
44
45 return s_instance.get();
46 }
47
48 private:
49 static std::unique_ptr<Type> s_instance; ///< Single instance of the Type class within the project
50 static std::once_flag s_init_flag; ///< Flag indicating whether #Get_Instance has already been called
51 };
